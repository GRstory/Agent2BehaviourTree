root :
    selector :
        # Mandatory Scan on Turn 1
        sequence :
            NOT HasScannedEnemy()
            condition : HasMP(15)
            task : Scan()

        # Proactive Defense against FrostTouch
        sequence :
            condition : EnemyIsTelegraphing(FrostTouch)
            task : Defend()

        # Proactive Defense against HeavySlam
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            task : Defend()

        # Cleanse Burn
        sequence :
            HasScannedEnemy()
            condition : EnemyHasBuff(Burn)
            condition : HasMP(25)
            task : Cleanse()

        # Cleanse AttackDown
        sequence :
            HasScannedEnemy()
            condition : EnemyHasBuff(AttackDown)
            condition : HasMP(25)
            task : Cleanse()

        # FireSpell if enemy is Ice
        sequence :
            condition : EnemyWeakTo(Ice)
            condition : HasMP(20)
            task : FireSpell()

        # IceSpell if enemy is Fire
        sequence :
            condition : EnemyWeakTo(Fire)
            condition : HasMP(20)
            task : IceSpell()

        # Default FireSpell after Scan
        sequence :
            HasScannedEnemy()
            condition : HasMP(20)
            task : FireSpell()

        # Heal if low HP
        sequence :
            condition : IsPlayerHPLow(70)
            condition : CanHeal()
            task : Heal()

        # Default: Attack
        task : Attack()
