root :
    selector :
        # Turn 1: Scan
        sequence :
            condition : IsTurnEarly(1)
            condition : HasMP(15)
            task : Scan()

        # React to HeavySlam telegraph
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            task : Defend()

        # Exploit Ice Weakness after Scan
        sequence :
            condition : HasScannedEnemy()
            condition : EnemyWeakTo(Ice)
            condition : HasMP(20)
            task : IceSpell()

        # Heal when critical
        sequence :
            condition : IsPlayerHPLow(40)
            condition : CanHeal()
            task : Heal()
        
        # Default: Attack
        task : Attack()
