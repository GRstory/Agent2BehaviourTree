root :
    selector :
        # Turn 1: Scan
        sequence :
            condition : IsTurnEarly(1)
            condition : HasMP(15)
            task : Scan()

        # React to HeavySlam telegraph
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            task : Defend()

        # Heal when critical
        sequence :
            condition : IsPlayerHPLow(30)
            condition : CanHeal()
            task : Heal()
        
        # Exploit Fire Weakness
        selector :
            sequence :
                condition : HasScannedEnemy()
                condition : EnemyWeakTo(Fire)
                condition : HasMP(20)
                task : FireSpell()
            
            # Default: Attack
            task : Attack()
