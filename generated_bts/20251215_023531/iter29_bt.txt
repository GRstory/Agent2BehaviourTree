root :
    selector :
        # Mandatory Scan on Turn 1
        sequence :
            NOT HasScannedEnemy()
            condition : HasMP(15)
            task : Scan()

        # Defend against debuffs
        sequence :
            condition : EnemyIsTelegraphing(Debuff)
            task : Defend()

        # Defend against HeavySlam
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            task : Defend()

        # IceSpell if enemy is Weak to Ice
        sequence :
            HasScannedEnemy()
            EnemyWeakTo(Ice)
            condition : HasMP(20)
            task : IceSpell()

        # FireSpell if enemy is Weak to Fire or Neutral
        sequence :
            HasScannedEnemy()
            EnemyWeakTo(Fire)
            condition : HasMP(20)
            task : FireSpell()

        # Cleanse Burn
        sequence :
            HasScannedEnemy()
            condition : EnemyHasBuff(Burn)
            condition : HasMP(25)
            task : Cleanse()

        # Cleanse AttackDown
        sequence :
            condition : EnemyHasBuff(AttackDown)
            condition : HasMP(25)
            task : Cleanse()

        # Heal if low HP
        sequence :
            condition : IsPlayerHPLow(40)
            condition : CanHeal()
            task : Heal()

        # Default: Attack
        task : Attack()
