root :
    selector :
        # Scan on Turn 1
        sequence :
            NOT HasScannedEnemy()
            condition : HasMP(15)
            task : Scan()

        # Defend against HeavySlam
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            task : Defend()
        
        # IceSpell if weak and scanned
        sequence :
            HasScannedEnemy()
            EnemyWeakTo(Fire)
            condition : HasMP(20)
            task : IceSpell()

        # Heal if low HP
        sequence :
            condition : IsPlayerHPLow(50)
            condition : CanHeal()
            task : Heal()

        # Cleanse Burn/AttackDown
        sequence :
            condition : HasMP(25)
            # Implied condition: Burn or AttackDown is active
            task : Cleanse()
        
        # Default: Attack
        task : Attack()
