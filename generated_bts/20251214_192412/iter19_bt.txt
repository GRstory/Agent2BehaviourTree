root :
    selector :
        # Defend against telegraphed HeavySlam
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            task : Defend()

        # Defend against IceWraith
        sequence :
            condition : EnemyWeakTo(Fire)
            task : Defend()

        # Scan on turn 1 if enemy is not already scanned
        sequence :
            condition : NOT HasScannedEnemy()
            condition : HasMP(15)
            task : Scan()

        # Immediately IceSpell after scanning if enemy is weak to Fire
        sequence :
            condition : HasScannedEnemy()
            condition : EnemyWeakTo(Fire)
            condition : HasMP(20)
            task : IceSpell()
        
        # IceSpell if enemy is weak to Fire
        sequence :
            condition : EnemyWeakTo(Fire)
            condition : HasMP(20)
            task : IceSpell()

        # FireSpell if enemy is weak to Ice
        sequence :
            condition : EnemyWeakTo(Ice)
            condition : HasMP(20)
            task : FireSpell()

        # Heal when critical
        sequence :
            condition : IsPlayerHPLow(70)
            condition : CanHeal()
            task : Heal()

        # Cleanse debuffs immediately
        sequence :
            condition : HasStatus(AttackDown)
            condition : HasMP(25)
            task : Cleanse()

        # Default: Attack
        task : Attack()
