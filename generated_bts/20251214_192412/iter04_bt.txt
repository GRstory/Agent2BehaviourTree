root :
    selector :
        # Scan on turn 1
        sequence :
            NOT HasScannedEnemy()
            condition : HasMP(15)
            task : Scan()

        # Cleanse debuffs
        sequence :
            condition : HasStatus(AttackDown)
            condition : HasMP(25)
            task : Cleanse()

        # Defend against Debuff
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            task : Defend()

        # Heal when critical
        sequence :
            condition : IsPlayerHPLow(30)
            condition : CanHeal()
            task : Heal()

        # FireSpell if enemy is weak to Ice
        sequence :
            condition : HasScannedEnemy()
            condition : EnemyWeakTo(Ice)
            condition : HasMP(20)
            task : FireSpell()
        
        # Default: Attack
        task : Attack()
