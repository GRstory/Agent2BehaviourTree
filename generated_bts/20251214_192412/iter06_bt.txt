root :
    selector :
        # Defend against HeavySlam
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            task : Defend()

        # Scan on turn 1
        sequence :
            NOT HasScannedEnemy()
            condition : HasMP(15)
            task : Scan()

        # Cleanse debuffs
        sequence :
            condition : HasStatus(AttackDown)
            condition : HasMP(25)
            task : Cleanse()

        # Heal when critical
        sequence :
            condition : IsPlayerHPLow(40)
            condition : CanHeal()
            task : Heal()

        # IceSpell if enemy is weak to Fire
        sequence :
            condition : HasScannedEnemy()
            condition : EnemyWeakTo(Fire)
            condition : HasMP(20)
            task : IceSpell()

        # FireSpell if enemy is weak to Ice
        sequence :
            condition : HasScannedEnemy()
            condition : EnemyWeakTo(Ice)
            condition : HasMP(20)
            task : FireSpell()
        
        # Default: Attack
        task : Attack()
