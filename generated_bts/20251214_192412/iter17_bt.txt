root :
    selector :
        # Defend against telegraphed HeavySlam
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            task : Defend()

        # Scan on turn 1
        sequence :
            condition : HasMP(15)
            task : Scan()

        # Immediately IceSpell after scanning
        sequence :
            condition : HasScannedEnemy()
            condition : EnemyWeakTo(Fire)
            condition : HasMP(20)
            task : IceSpell()

        # IceSpell if enemy is weak to Fire
        sequence :
            condition : EnemyWeakTo(Fire)
            condition : HasMP(20)
            task : IceSpell()

        # Heal when critical
        sequence :
            condition : IsPlayerHPLow(50)
            condition : CanHeal()
            task : Heal()

        # Cleanse debuffs immediately
        sequence :
            condition : HasStatus(AttackDown)
            condition : HasMP(25)
            task : Cleanse()

        # Default: Attack
        task : Attack()
