root :
    selector :
        # Scan on turn 1
        sequence :
            NOT HasScannedEnemy()
            condition : HasMP(15)
            task : Scan()

        # Defend against HeavySlam
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            task : Defend()

        # Heal when critical
        sequence :
            condition : IsPlayerHPLow(30)
            condition : CanHeal()
            task : Heal()

        # IceSpell if enemy is weak to Ice
        sequence :
            condition : HasScannedEnemy()
            condition : EnemyWeakTo(Ice)
            condition : HasMP(20)
            task : IceSpell()
        
        # Default: Attack
        task : Attack()
