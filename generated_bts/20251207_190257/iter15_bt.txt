root :
    selector :
        # Early Scan
        sequence :
            condition : IsTurnEarly(1)
            task : Scan()

        # Ice Spell after Scan
        sequence :
            condition : HasScannedEnemy()
            condition : EnemyWeakTo(Ice)
            condition : HasTP(10)
            task : IceSpell()

        # Proactive Defend & Counter Attack
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            condition : IsPlayerHPLow(60)
            task : Defend()

        # Emergency Heal
        sequence :
            condition : IsPlayerHPLow(80)
            condition : CanHeal()
            task : Heal()

        # Exploit weakness
        selector :
            sequence :
                condition : HasScannedEnemy()
                condition : EnemyWeakTo(Fire)
                condition : HasMP(20)
                condition : HasTP(10)
                task : FireSpell()
            sequence :
                condition : HasScannedEnemy()
                condition : EnemyWeakTo(Lightning)
                condition : HasMP(20)
                condition : HasTP(10)
                task : LightningSpell()

        # Attack every 3 turns
        sequence :
            condition : IsTurnEarly(3)
            task : Attack()

        # Default: Attack
        task : Attack()
