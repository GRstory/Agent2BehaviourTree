root :
    selector :
        # Reactive Defense
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            task : Defend()

        # Scan on turn 1 and 5
        selector :
            sequence :
                condition : IsTurnEarly(1)
                task : Scan()
            sequence :
                condition : IsTurnEarly(5)
                task : Scan()

        # Prioritize IceSpell when enemy is weak
        sequence :
            condition : HasScannedEnemy()
            condition : EnemyWeakTo(Ice)
            condition : HasMP(20)
            condition : HasTP(20)
            task : IceSpell()

        # Heal when player is low and enemy is strong
        sequence :
            condition : IsPlayerHPLow(30)
            condition : IsEnemyHPLow(60)
            condition : CanHeal()
            task : Heal()

        # Exploit weakness
        selector :
            sequence :
                condition : HasScannedEnemy()
                condition : EnemyWeakTo(Fire)
                condition : HasMP(20)
                condition : HasTP(20)
                task : FireSpell()
            sequence :
                condition : HasScannedEnemy()
                condition : EnemyWeakTo(Lightning)
                condition : HasMP(20)
                condition : HasTP(20)
                task : LightningSpell()

        # Default: Attack
        task : Attack()
