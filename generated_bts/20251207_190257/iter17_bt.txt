root :
    selector :
        # Scan every 3 turns, starting turn 1
        sequence :
            condition : IsTurnEarly(1)
            task : Scan()
        sequence :
            condition : IsTurnEarly(4)
            task : Scan()

        # Proactive Defend - Low HP
        sequence :
            condition : IsPlayerHPLow(40)
            task : Defend()

        # Defend Telegraph
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            condition : IsPlayerHPLow(60)
            task : Defend()

        # Exploit weakness
        selector :
            sequence :
                condition : HasScannedEnemy()
                condition : EnemyWeakTo(Fire)
                condition : HasMP(20)
                task : FireSpell()
            sequence :
                condition : HasScannedEnemy()
                condition : EnemyWeakTo(Ice)
                condition : HasMP(20)
                task : IceSpell()
            sequence :
                condition : HasScannedEnemy()
                condition : EnemyWeakTo(Lightning)
                condition : HasMP(20)
                task : LightningSpell()

        # Power Strike if possible
        sequence :
            condition : HasTP(30)
            task : PowerStrike()

        # Emergency Heal
        sequence :
            condition : IsPlayerHPLow(80)
            condition : CanHeal()
            task : Heal()

        # Default: Attack
        task : Attack()
