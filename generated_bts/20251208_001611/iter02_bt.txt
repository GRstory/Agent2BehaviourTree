root :
    selector :
        # Defensive Priority
        task : Defend()

        # Heal when critical
        sequence :
            condition : IsPlayerHPLow(20)
            condition : CanHeal()
            task : Heal()
        
        # Exploit Weakness
        sequence :
            condition : HasScannedEnemy()
            condition : EnemyWeakTo(Ice)
            condition : HasMP(20)
            task : FireSpell()
        
        # Scan for Weakness
        sequence :
            condition : HasMP(15)
            condition : IsTurnEarly(1)
            task : Scan()
        
        # Default: Attack
        task : Attack()
