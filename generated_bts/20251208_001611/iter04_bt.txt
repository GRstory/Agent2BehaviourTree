root :
    selector :
        # Mandatory Scan on Turn 1
        sequence :
            condition : IsTurnEarly(1)
            condition : HasMP(15)
            task : Scan()

        # Exploit Weakness
        sequence :
            condition : HasScannedEnemy()
            condition : EnemyWeakTo(Ice)
            task : IceSpell()

        # Heal when critical
        sequence :
            condition : IsPlayerHPLow(20)
            condition : CanHeal()
            task : Heal()

        # Scan for Weakness (if not already scanned)
        sequence :
            condition : HasMP(20)
            condition : HasScannedEnemy()
            task : Scan()

        # Initial Defense
        task : Defend()

        # Default: Attack
        task : Attack()
