root :
    selector :
        # Prioritized Defense against Telegraphing
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            task : Defend()

        # Immediate Ice Spell Trigger on Scan and weakness
        sequence :
            condition : EnemyWeakTo(Ice)
            task : IceSpell()

        # Prioritized Heal at lower HP threshold
        sequence :
            condition : IsPlayerHPLow(30)
            condition : CanHeal()
            task : Heal()

        # Prioritized Fire Spell Usage
        sequence :
            condition : HasScannedEnemy()
            condition : EnemyWeakTo(Fire)
            condition : HasMP(20)
            task : FireSpell()

        # Prioritized Lightning Spell Usage
        sequence :
            condition : HasScannedEnemy()
            condition : EnemyWeakTo(Lightning)
            condition : HasMP(20)
            task : LightningSpell()

        # Conditional Scan Trigger on turn 1
        sequence :
            condition : IsTurnEarly(2)
            condition : HasMP(15)
            task : Scan()

        # Default: Attack
        task : Attack()
