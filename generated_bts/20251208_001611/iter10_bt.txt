root :
    selector :
        # Prioritized Ice Spell Usage
        sequence :
            condition : HasScannedEnemy()
            condition : EnemyWeakTo(Ice)
            condition : HasMP(20)
            task : IceSpell()

        # Turn 1: Scan
        sequence :
            condition : IsTurnEarly(1)
            condition : HasMP(15)
            task : Scan()

        # Proactive Ice Spell on Turn 1
        sequence :
            condition : IsTurnEarly(1)
            condition : HasMP(20)
            task : IceSpell()

        # Heal when critical, with shorter telegraph
        sequence :
            condition : IsPlayerHPLow(20)
            condition : CanHeal()
            task : Heal()

        # Proactive Defend
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            task : Defend()

        # Damage Burst on telegraph
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            condition : HasTP(30)
            task : PowerStrike()

        # Default: Attack
        task : Attack()
