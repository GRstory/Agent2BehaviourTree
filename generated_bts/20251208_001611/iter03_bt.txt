root :
    selector :
        # Initial Defense
        task : Defend()

        # Heal when critical
        sequence :
            condition : IsPlayerHPLow(40)
            condition : CanHeal()
            task : Heal()

        # Exploit Weakness
        sequence :
            condition : HasScannedEnemy()
            condition : EnemyWeakTo(Ice)
            task : IceSpell()
            
        # Scan for Weakness
        sequence :
            condition : HasMP(15)
            task : Scan()

        # Default: Attack
        task : Attack()
