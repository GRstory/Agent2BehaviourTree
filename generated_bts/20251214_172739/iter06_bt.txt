root :
    selector :
        # Defend against HeavySlam
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            task : Defend()

        # Scan on Turn 1
        sequence :
            NOT condition : HasScannedEnemy()
            condition : HasMP(15)
            task : Scan()

        # Cleanse if burned
        sequence :
            condition : HasStatus(BURN)
            condition : HasMP(25)
            task : Cleanse()

        # Heal when critical
        sequence :
            condition : IsPlayerHPLow(50)
            condition : CanHeal()
            task : Heal()

        # IceSpell if enemy weak and affordable
        sequence :
            condition : EnemyWeakTo(Fire)
            condition : HasMP(20)
            task : IceSpell()

        # Default: Attack
        task : Attack()
