root :
    selector :
        # Scan on Turn 1
        sequence :
            NOT condition : HasScannedEnemy()
            condition : HasMP(15)
            task : Scan()

        # Defend against HeavySlam
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            task : Defend()

        # Heal when critical
        sequence :
            condition : IsPlayerHPLow(30)
            condition : CanHeal()
            task : Heal()
        
        # IceSpell if enemy weak and affordable
        sequence :
            condition : EnemyWeakTo(Fire)
            condition : HasMP(20)
            task : IceSpell()

        # Default: Attack
        task : Attack()
