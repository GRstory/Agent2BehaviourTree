root :
    selector :
        # Defend against HeavySlam, FlameStrike, Debuff, or FrostTouch
        sequence :
            selector :
                condition : EnemyIsTelegraphing(HeavySlam)
                condition : EnemyIsTelegraphing(FlameStrike)
                condition : EnemyIsTelegraphing(Debuff)
                condition : EnemyIsTelegraphing(FrostTouch)
            task : Defend()

        # Heal when HP is low
        sequence :
            condition : IsPlayerHPLow(40)
            condition : CanHeal()
            task : Heal()

        # Exploit Fire weakness
        sequence :
            condition : EnemyHasElement(Fire)
            condition : HasMP(20)
            task : IceSpell()

        # Exploit Ice weakness
        sequence :
            condition : EnemyHasElement(Ice)
            condition : HasMP(20)
            task : FireSpell()

        # Default: Attack
        task : Attack()
