root :
    selector :
        # Heal when critical
        sequence :
            condition : IsPlayerHPLow(50)
            condition : CanHeal()
            task : Heal()

        # Defend against Debuff
        sequence :
            condition : EnemyIsTelegraphing(Debuff)
            task : Defend()

        # Defend against FrostTouch
        sequence :
            condition : EnemyIsTelegraphing(FrostTouch)
            task : Defend()

        # Exploit Ice weakness
        sequence :
            condition : EnemyHasElement(Fire)
            condition : HasMP(20)
            task : IceSpell()

        # Exploit Fire weakness
        sequence :
            condition : EnemyHasElement(Ice)
            condition : HasMP(20)
            task : FireSpell()

        # Defend against HeavySlam Telegraph
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            task : Defend()
        
        # Default: Attack
        task : Attack()
