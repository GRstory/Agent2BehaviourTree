root :
    selector :
        # Defend against Debuff
        sequence :
            condition : EnemyIsTelegraphing(Debuff)
            task : Defend()

        # Heal when HP is low
        sequence :
            condition : IsPlayerHPLow(40)
            condition : CanHeal()
            task : Heal()

        # FireSpell when enemy has Ice
        sequence :
            condition : EnemyHasElement(Ice)
            condition : HasMP(20)
            task : FireSpell()

        # IceSpell when enemy has Fire
        sequence :
            condition : EnemyHasElement(Fire)
            condition : HasMP(20)
            task : IceSpell()

        # Use FireSpell if enough MP
        sequence :
            condition : HasMP(20)
            task : FireSpell()
        
        # Default: Attack
        task : Attack()
