root :
    selector :
        # Defend against HeavySlam
        sequence :
            condition : EnemyIsTelegraphing(HeavySlam)
            task : Defend()

        # Defend against Debuff
        sequence :
            condition : EnemyIsTelegraphing(Debuff)
            task : Defend()

        # Defend against FrostBlast
        sequence :
            condition : EnemyIsTelegraphing(FrostBlast)
            task : Defend()

        # FireSpell when enemy has Ice
        sequence :
            condition : EnemyHasElement(Ice)
            condition : HasMP(20)
            task : FireSpell()

        # IceSpell when enemy has Fire
        sequence :
            condition : EnemyHasElement(Fire)
            condition : HasMP(20)
            task : IceSpell()

        # Heal when HP is low
        sequence :
            condition : IsPlayerHPLow(40)
            condition : CanHeal()
            task : Heal()

        # Attack when enemy is Neutral
        sequence :
            condition : EnemyHasElement(Neutral)
            task : Attack()

        # Default: Attack
        task : Attack()
